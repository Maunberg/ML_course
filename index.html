NB

Принцип работы

Использует формулу Байеса и предположение о независимости признаков внутри класса. Выбирается класс с максимальной апостериорной вероятностью.

Собственная реализация без sklearn

import numpy as np

class GaussianNBFromScratch:
    def fit(self, X, y):
        self.classes_ = np.unique(y)
        self.mean_ = {}
        self.var_ = {}
        self.prior_ = {}
        for c in self.classes_:
            Xc = X[y == c]
            self.mean_[c] = Xc.mean(axis=0)
            self.var_[c] = Xc.var(axis=0) + 1e-9
            self.prior_[c] = len(Xc) / len(X)

    def _log_likelihood(self, x, c):
        mean = self.mean_[c]
        var = self.var_[c]
        return -0.5 * np.sum(np.log(2 * np.pi * var) + ((x - mean) ** 2) / var)

    def predict(self, X):
        preds = []
        for x in X:
            scores = [np.log(self.prior_[c]) + self._log_likelihood(x, c) for c in self.classes_]
            preds.append(self.classes_[np.argmax(scores)])
        return np.array(preds)

Реализация sklearn

from sklearn.naive_bayes import GaussianNB

model = GaussianNB()
model.fit(X_train, y_train)
y_pred = model.predict(X_test)

Гиперпараметры sklearn

  • var_smoothing: стабилизация дисперсии для численной устойчивости.
  • priors: априорные вероятности классов вручную.

Готовое set_params для Grid Search

    "var_smoothing": np.logspace(-12, -6, 7)

KNN

Принцип работы

Для нового объекта ищет ближайшие точки в обучающей выборке и голосует по их классам. Качество сильно зависит от масштаба признаков.

Собственная реализация без sklearn

import numpy as np
from collections import Counter

class KNNFromScratch:
    def __init__(self, k=5):
        self.k = k

    def fit(self, X, y):
        self.X_train = X
        self.y_train = y

    def _distance(self, a, b):
        return np.sqrt(np.sum((a - b) ** 2))

    def predict(self, X):
        preds = []
        for x in X:
            dists = [self._distance(x, x_train) for x_train in self.X_train]
            idx = np.argsort(dists)[:self.k]
            nearest_labels = self.y_train[idx]
            preds.append(Counter(nearest_labels).most_common(1)[0][0])
        return np.array(preds)

Реализация sklearn

from sklearn.neighbors import KNeighborsClassifier

model = KNeighborsClassifier(n_neighbors=5)
model.fit(X_train, y_train)
y_pred = model.predict(X_test)

Гиперпараметры sklearn

  • n_neighbors: число соседей.
  • weights: равные веса или по расстоянию.
  • metric: метрика расстояния (euclidean, manhattan).
  • p: степень нормы Минковского (2 - Евклид, 1 - Манхэттен).

RF

Принцип работы

Собирает ансамбль деревьев по бутстрэп-выборкам и случайным подмножествам признаков. Итог - голосование деревьев, что снижает переобучение.

Собственная реализация без sklearn

import numpy as np
from collections import Counter

class TinyDecisionStump:
    def fit(self, X, y):
        self.feature_ = 0
        self.threshold_ = np.median(X[:, 0])
        left = y[X[:, 0] <= self.threshold_]
        right = y[X[:, 0] > self.threshold_]
        self.left_class_ = Counter(left).most_common(1)[0][0]
        self.right_class_ = Counter(right).most_common(1)[0][0]

    def predict(self, X):
        return np.where(X[:, 0] <= self.threshold_, self.left_class_, self.right_class_)

class RandomForestToy:
    def __init__(self, n_estimators=10):
        self.n_estimators = n_estimators
        self.trees = []

    def fit(self, X, y):
        n = len(X)
        self.trees = []
        for _ in range(self.n_estimators):
            idx = np.random.choice(n, n, replace=True)
            tree = TinyDecisionStump()
            tree.fit(X[idx], y[idx])
            self.trees.append(tree)

    def predict(self, X):
        preds = np.array([tree.predict(X) for tree in self.trees]).T
        return np.array([Counter(row).most_common(1)[0][0] for row in preds])

Реализация sklearn

from sklearn.ensemble import RandomForestClassifier

model = RandomForestClassifier(random_state=42)
model.fit(X_train, y_train)
y_pred = model.predict(X_test)

Гиперпараметры sklearn

  • n_estimators: количество деревьев.
  • max_depth: ограничение глубины дерева.
  • min_samples_split, min_samples_leaf: контроль ветвления.
  • max_features: сколько признаков пробовать на каждом сплите.
  • bootstrap: использовать ли бутстрэп.

DT

Принцип работы

Рекурсивно делит данные по признакам, максимизируя прирост информативности (например, по Gini). Легко интерпретируется, но склонно к переобучению.

Собственная реализация без sklearn

import numpy as np

class DecisionTreeOneSplit:
    def fit(self, X, y):
        self.feature_ = 0
        self.threshold_ = np.median(X[:, 0])
        left = y[X[:, 0] <= self.threshold_]
        right = y[X[:, 0] > self.threshold_]
        self.left_class_ = np.bincount(left).argmax()
        self.right_class_ = np.bincount(right).argmax()

    def predict(self, X):
        return np.where(X[:, 0] <= self.threshold_, self.left_class_, self.right_class_)

Реализация sklearn

from sklearn.tree import DecisionTreeClassifier

model = DecisionTreeClassifier(random_state=42)
model.fit(X_train, y_train)
y_pred = model.predict(X_test)

Гиперпараметры sklearn

  • criterion: функция качества (gini или entropy).
  • max_depth: глубина дерева.
  • min_samples_split, min_samples_leaf: регуляризация.
  • max_features: ограничение числа признаков для сплита.
  • ccp_alpha: post-pruning через cost complexity pruning.

Готовое set_params для Grid Search

LR

Принцип работы

Оценивает вероятность класса через сигмоиду от линейной комбинации признаков. Граница решения линейная, обучение обычно через градиентные методы.

Собственная реализация без sklearn

import numpy as np

class LogisticRegressionFromScratch:
    def __init__(self, lr=0.05, n_iter=2000):
        self.lr = lr
        self.n_iter = n_iter

    def _sigmoid(self, z):
        return 1 / (1 + np.exp(-z))

    def fit(self, X, y):
        n, m = X.shape
        self.w = np.zeros(m)
        self.b = 0
        for _ in range(self.n_iter):
            z = X @ self.w + self.b
            y_hat = self._sigmoid(z)
            dw = (1 / n) * (X.T @ (y_hat - y))
            db = (1 / n) * np.sum(y_hat - y)
            self.w -= self.lr * dw
            self.b -= self.lr * db

    def predict(self, X):
        probs = self._sigmoid(X @ self.w + self.b)
        return (probs >= 0.5).astype(int)

Реализация sklearn

from sklearn.linear_model import LogisticRegression

model = LogisticRegression(max_iter=2000)
model.fit(X_train, y_train)
y_pred = model.predict(X_test)

Гиперпараметры sklearn

  • C: сила регуляризации (меньше значение - сильнее регуляризация).
  • penalty: тип регуляризации (l1, l2, elasticnet).
  • solver: алгоритм оптимизации (liblinear, lbfgs, saga).
  • class_weight: веса классов при дисбалансе.
  • max_iter: лимит итераций оптимизатора.
